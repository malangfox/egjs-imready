Type Description

Example
import { useImReady } from "@egjs/react-imready";

function App() {
  const {
        register,
        isReady,
        isPreReady,
        preReadyCount,
        readyCount,
        totalCount,
    } = useImReady({
        selector: "img",
    });
    // <div ref={register()}></div>
